### Q1: Vue 3 有哪些新特性？

**参考答案：**

**Vue 3 新特性一览：**

| 特性 | 说明 |
|------|------|
| **Composition API** | 组合式 API，更灵活的代码组织 |
| **Teleport** | 传送门组件 |

| **更小的体积** | Tree-shaking 支持 |
| **自定义渲染器** | 支持自定义渲染 |

**核心变化：**

```javascript
// Vue 2 vs Vue 3
// Vue 2 - Options API
export default {
  data() { return { count: 0 } },
  methods: { increment() { this.count++ } }
}

// Vue 3 - Composition API
import { ref } from 'vue'

export default {
  setup() {
    const count = ref(0)
    const increment = () => count.value++
    return { count, increment }
  }
}
```

### Q3: ref 和 reactive 的区别？

**参考答案：**

**ref vs reactive：**

```javascript
import { ref, reactive } from 'vue'

// ref: 适合基本类型
const count = ref(0)
console.log(count.value) // 访问值
count.value++ // 修改值

// reactive: 适合对象类型
const state = reactive({
  count: 0,
  name: 'Vue'
})
console.log(state.count) // 直接访问
state.count++ // 直接修改

// ref 也支持对象
const state = ref({ count: 0 })
console.log(state.value.count) // 需要 .value
```

**使用场景：**

```javascript
// 使用 ref
const message = ref('') // 字符串
const isLoading = ref(false) // 布尔
const count = ref(0) // 数字

// 使用 reactive
const form = reactive({
  username: '',
  password: '',
  remember: false
})

const user = reactive({
  profile: {
    name: '',
    avatar: ''
  },
  settings: {
    theme: 'dark'
  }
})
```

### Q4: toRef 和 toRefs 的作用？

**参考答案：**

**toRef 和 toRefs：**

```javascript
import { reactive, toRef, toRefs } from 'vue'

// 原始 reactive 对象
const state = reactive({
  name: 'Vue',
  version: '3.0',
  author: 'Evan You'
})

// toRef: 从 reactive 对象创建 ref
const name = toRef(state, 'name')
console.log(name.value) // 'Vue'
name.value = 'Vue 3' // 修改会影响原对象

// toRefs: 将 reactive 对象转为普通对象（每个属性都是 ref）
const stateRefs = toRefs(state)
// 结果：
// {
//   name: Ref<'Vue'>,
//   version: Ref<'3.0'>,
//   author: Ref<'Evan You'>
// }

// 解构后保持响应式
const { name, version } = toRefs(state)
console.log(name.value) // 'Vue'
```

**使用场景：**

```javascript
// 模板中使用
const state = reactive({
  title: 'RAG 问答系统',
  count: 0
})

// 模板中使用 toRefs 返回的对象
// 不需要使用 .value
const { title, count } = toRefs(state)

// 在 setup 返回时使用
return {
  // 保持响应式
  ...toRefs(state),
  
  // 或者单独使用
  title: toRef(state, 'title')
}
```

### Q5: computed 和 watch 的区别？

**参考答案：**

**computed 计算属性：**

```javascript
import { ref, computed } from 'vue'

const firstName = ref('John')
const lastName = ref('Doe')

// 计算属性 - 基于响应式数据计算得出
const fullName = computed(() => {
  return `${firstName.value} ${lastName.value}`
})

console.log(fullName.value) // 'John Doe'
firstName.value = 'Jane'
console.log(fullName.value) // 'Jane Doe'

// 带 setter 的计算属性
const fullNameWritable = computed({
  get: () => `${firstName.value} ${lastName.value}`,
  set: (val) => {
    const [first, last] = val.split(' ')
    firstName.value = first
    lastName.value = last
  }
})
```

**watch 侦听器：**

```javascript
import { ref, watch } from 'vue'

const count = ref(0)

// 侦听 ref
watch(count, (newValue, oldValue) => {
  console.log(`count 从 ${oldValue} 变为 ${newValue}`)
})

count.value++

// 侦听多个数据源
const firstName = ref('')
const lastName = ref('')

watch([firstName, lastName], ([newFirst, newLast], [oldFirst, oldLast]) => {
  console.log(`名字从 ${oldFirst} ${oldLast} 变为 ${newFirst} ${newLast}`)
})

// 侦听 reactive 对象
const state = reactive({ count: 0 })
watch(() => state.count, (newVal, oldVal) => {
  console.log(`count 变为 ${newVal}`)
})

// immediate 选项
watch(count, (newVal) => {
  console.log('立即执行', newVal)
}, { immediate: true })
```

**区别对比：**

| 特性 | computed | watch |
|------|----------|-------|
| 用途 | 计算属性 | 响应副作用 |
| 缓存 | 有缓存机制 | 无缓存 |
| 返回值 | 返回值 | 不返回值 |
| 触发 | 依赖变化时触发 | 手动指定 |

## 2. Composition API

### Q6: Composition API 的优势是什么？

**参考答案：**

**Composition API 优势：**

```javascript
// 1. 更好的逻辑复用
// Vue 2: Mixins 有命名冲突、来源不清问题
const myMixin = {
  created() { console.log('mixin created') }
}

// Vue 3: Composables 清晰可控
function useLogger() {
  const log = (msg) => console.log(msg)
  onMounted(() => log('component mounted'))
  return { log }
}

// 2. 更好的类型推导
import { ref, computed } from 'vue'
const count = ref(0) // 自动推导类型

// 3. 更好的代码组织
// 相关逻辑放在一起
function useSearch() {
  const query = ref('')
  const results = ref([])
  const isLoading = ref(false)
  
  const search = async () => {
    isLoading.value = true
    results.value = await api.search(query.value)
    isLoading.value = false
  }
  
  watch(query, search)
  
  return { query, results, isLoading, search }
}
```

**对比 Options API：**

```javascript
// Options API - 逻辑分散
export default {
  data() { return { query: '', results: [] } },
  computed: { filteredResults() { ... } },
  watch: { query() { this.search() } },
  methods: { async search() { ... } },
  mounted() { this.loadData() }
}

// Composition API - 逻辑聚合
export default {
  setup() {
    // 搜索相关逻辑
    const { query, results, search } = useSearch()
    
    // 加载相关逻辑
    const { data, load } = useDataLoader()
    
    onMounted(load)
    
    return { query, results, data }
  }
}
```

### Q7: 如何使用 use... 命名约定（Composables）？

**参考答案：**

**Composables 模式：**

```javascript
// useSearch.js
import { ref, computed } from 'vue'

export function useSearch() {
  const query = ref('')
  const results = ref([])
  const isLoading = ref(false)
  
  const hasResults = computed(() => results.value.length > 0)
  
  const search = async () => {
    if (!query.value) {
      results.value = []
      return
    }
    
    isLoading.value = true
    try {
      results.value = await api.search(query.value)
    } finally {
      isLoading.value = false
    }
  }
  
  // 自动防抖
  let timeout
  watch(query, () => {
    clearTimeout(timeout)
    timeout = setTimeout(search, 300)
  })
  
  return {
    query,
    results,
    isLoading,
    hasResults,
    search
  }
}

// 组件中使用
import { useSearch } from '@/composables/useSearch'

export default {
  setup() {
    const { query, results, isLoading, hasResults } = useSearch()
    
    return { query, results, isLoading, hasResults }
  }
}
```

### Q8: 如何处理组件间的逻辑复用？

**参考答案：**

```javascript
// 方案1: Mixins（不推荐）
const myMixin = {
  data() { return { count: 0 } },

}

// 方案2: Composables（推荐）
function useCounter(initialCount = 0) {
  const count = ref(initialCount)
  
  const increment = () => count.value++
  const decrement = () => count.value--
  const reset = () => count.value = initialCount
  
  return { count, increment, decrement, reset }
}

// 方案3: Vuex/Pinia（全局状态）
// store/counter.js
import { defineStore } from 'pinia'

export const useCounterStore = defineStore('counter', {
  state: () => ({ count: 0 }),
  actions: {
    increment() { this.count++ },
    decrement() { this.count-- }
  }
})
```

| 跨组件复用 | Composables |
| 全局应用状态 | Pinia |
| 页面级别状态 | Composables |

### Q9: 如何在 setup 中使用生命周期钩子？

**参考答案：**

**生命周期钩子：**

```javascript
import { 
  onMounted, 
  onUpdated, 
  onUnmounted,
  onBeforeMount,
  onBeforeUpdate,
  onBeforeUnmount,
  onErrorCaptured,
  onRenderTracked,
  onRenderTriggered
} from 'vue'

export default {
  setup() {
    // 挂载前
    onBeforeMount(() => {
      console.log('onBeforeMount')
    })
    
    // 挂载完成
    onMounted(() => {
      console.log('onMounted')
      // 适合 DOM 操作、绑定事件
    })
    
    // 更新前
    onBeforeUpdate(() => {
      console.log('onBeforeUpdate')
    })
    
    // 更新完成
    onUpdated(() => {
      console.log('onUpdated')
      // 避免在此修改状态，可能导致无限循环
    })
    
    // 卸载前
    onBeforeUnmount(() => {
      console.log('onBeforeUnmount')
      // 适合清理工作：取消订阅、清除定时器
    })
    
    // 卸载完成
    onUnmounted(() => {
      console.log('onUnmounted')
    })
    
    // 错误捕获
    onErrorCaptured((err, instance, info) => {
      console.error('错误:', err, info)
    })
  }
}
```

## 3. 响应式原理

### Q11: Vue 3 的响应式原理是什么？

**参考答案：**

**响应式原理：**

```javascript
// Vue 3 使用 Proxy 实现响应式

// ref 实现
function ref(initialValue) {
  return new Proxy({
    value: initialValue
  }, {
    get(target, key) {
      if (key === 'value') {
        track() // 收集依赖
      }
      return target[key]
    },
    set(target, key, newValue) {
      target[key] = newValue
      if (key === 'value') {
        trigger() // 触发更新
      }
      return true
    }
  })
}

// reactive 实现
function reactive(obj) {
  return new Proxy(obj, {
    get(target, key) {
      track(target, key)
      const value = target[key]
      // 深层响应式
      return typeof value === 'object' ? reactive(value) : value
    },
    set(target, key, newValue) {
      target[key] = newValue
      trigger(target, key)
      return true
    }
  })
}
```

**响应式追踪流程：**

```
数据变化
    │
    ▼
trigger() 触发更新
    │
    ▼
scheduler 调度更新
    │
    ▼
queueWatcher 队列更新
    │
    ▼
执行 Watcher.run()
    │
    ▼
组件重新渲染
```

### Q12: watchEffect 和 watch有什么区别？

**参考答案：**

**watchEffect：**

```javascript
import { ref, watchEffect } from 'vue'

const count = ref(0)

// 自动追踪依赖
watchEffect(() => {
  // 立即执行
  // 自动追踪 count 变化
  console.log('count 变为:', count.value)
})

count.value++ // 触发

// 可获取清理函数
watchEffect((onInvalidate) => {
  const timer = setTimeout(() => {
    console.log('5秒后执行')
  }, 5000)
  
  // 清理函数
  onInvalidate(() => {
    clearTimeout(timer)
  })
})
```

**watch vs watchEffect：**

| 特性 | watch | watchEffect |
|------|--------|-------------|
| 触发时机 | 指定数据变化时 | 立即执行 |
| 依赖追踪 | 手动指定 | 自动追踪 |
| 访问旧值 | 可以 | 不行 |
| 首次执行 | 需要 immediate | 自动执行 |
| 适用场景 | 条件性监听 | 副作用执行 |

### Q13: 如何避免响应式丢失？

      return payload !== null
    }

// 触发事件
setup(props, { emit }) {

    emit('click', { id: 1 })

### Q19: Pinia 相比 Vuex 有哪些优势？

**参考答案：**

**Pinia vs Vuex：**

| 特性 | Pinia | Vuex |
|------|-------|------|
| API | 简洁 | 较复杂 |
| TypeScript | 原生支持 | 需要适配 |

import { ref } from 'vue'

export function useStreamChat() {
  const messages = ref([])
  const isLoading = ref(false)
  
  async function sendMessage(question) {
    isLoading.value = true
    messages.value.push({
      role: 'user',
      content: question
    })
    
    try {
      const response = await fetch('/api/qa/ask/stream', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ question })
      })
      
      const reader = response.body.getReader()
      const decoder = new TextDecoder()
      let answer = ''
      
      while (true) {
        const { done, value } = await reader.read()
        if (done) break
        
        const text = decoder.decode(value)
        const lines = text.split('\n')
        
        for (const line of lines) {
          if (line.startsWith('data: ')) {
            const data = JSON.parse(line.slice(6))
            
            if (data.type === 'token') {
              answer += data.content
              // 更新最后一条消息
              const lastMsg = messages.value[messages.value.length - 1]
              if (lastMsg.role === 'assistant') {
                lastMsg.content = answer
              } else {
                messages.value.push({ role: 'assistant', content: answer })
              }
            } else if (data.type === 'sources') {
              // 处理来源
            }
          }
        }
      }
    } finally {
      isLoading.value = false
    }
  }
  
  return { messages, isLoading, sendMessage }
}

### Q23: 如何实现表单验证？

**参考答案：**

**表单验证实现：**

```javascript
// composables/useForm.js
import { ref, computed } from 'vue'

export function useFormValidation(initialData, rules) {
  const form = ref({ ...initialData })
  const errors = ref({})
  
  const validate = () => {
    errors.value = {}
    let isValid = true
    
    for (const [field, ruleList] of Object.entries(rules)) {
      for (const rule of ruleList) {
        const value = form.value[field]
        
        if (rule.required && !value) {
          errors.value[field] = rule.message || `${field}不能为空`
          isValid = false
          break
        }
        
        if (rule.pattern && !rule.pattern.test(value)) {
          errors.value[field] = rule.message || `${field}格式不正确`
          isValid = false
          break
        }
        
        if (rule.validator && !rule.validator(value)) {
          errors.value[field] = rule.message || `${field}验证失败`
          isValid = false
          break
        }
      }
    }
    
    return isValid
  }
  
  const reset = () => {
    form.value = { ...initialData }
    errors.value = {}
  }
  
  return { form, errors, validate, reset }
}

// 使用
const { form, errors, validate } = useFormValidation(
  { username: '', password: '' },
  {
    username: [
      { required: true, message: '请输入用户名' },
      { pattern: /^\w{4,}$/, message: '用户名至少4位' }
    ],
    password: [
      { required: true, message: '请输入密码' },
      { min: 6, message: '密码至少6位' }
    ]
  }
)
```
