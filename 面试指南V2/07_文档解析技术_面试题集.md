import chardet

def detect_encoding(file_path: str) -> str:
    """检测文件编码"""
    with open(file_path, 'rb') as f:
        raw_data = f.read(10000)  # 读取前10KB
    
    result = chardet.detect(raw_data)

def read_file_with_encoding(file_path: str) -> str:
    """根据检测到的编码读取文件"""
    encoding = detect_encoding(file_path)
    
    # 常见编码映射
    encoding_map = {
        'gb2312': 'gbk',
        'gb18030': 'gbk',
        'ascii': 'utf-8',
        None: 'utf-8',
    }
    
    encoding = encoding_map.get(encoding, encoding) or 'utf-8'
    
    with open(file_path, 'r', encoding=encoding, errors='replace') as f:
        return f.read()

def normalize_text(text: str) -> str:
    """规范化文本"""
    # 移除零宽字符
    text = re.sub(r'[\x00-\x08\x0b\x0c\x0e-\x1f\x7f]', '', text)
    
    # 规范化换行
    text = text.replace('\r\n', '\n').replace('\r', '\n')
    
    # 移除多余空白
    text = re.sub(r'[ \t]+', ' ', text)
    text = re.sub(r'\n{3,}', '\n\n', text)

### Q4: 如何验证文件的有效性？

**参考答案：**

**文件验证策略：**

```python
from pathlib import Path
import magic

class FileValidator:
    """文件验证器"""
    
    @staticmethod
    def validate_file(file_path: str) -> dict:
        """验证文件有效性"""
        result = {
            "valid": True,
            "errors": [],
            "warnings": []
        }
        
        path = Path(file_path)
        
        # 检查文件是否存在
        if not path.exists():
            result["valid"] = False
            result["errors"].append("文件不存在")
            return result
        
        # 检查文件大小
        if path.stat().st_size == 0:
            result["valid"] = False
            result["errors"].append("文件为空")
        
        if path.stat().st_size > 100 * 1024 * 1024:  # 100MB
            result["warnings"].append("文件过大")
        
        # 检查文件类型（MIME检测）
        mime = magic.from_file(str(path), mime=True)
        expected_types = {
            "pdf": "application/pdf",
            "docx": "application/vnd.openxmlformats-officedocument.wordprocessingml.document",
            "txt": "text/plain",
            "md": "text/plain",
        }
        
        return result
    
    @staticmethod
    def validate_content(content: str, min_length: int = 10) -> bool:
        """验证内容有效性"""
        if not content or not content.strip():
            return False
        
        if len(content.strip()) < min_length:
            return False
        
        # 检查是否全是空白字符
        if content.isspace():
            return False
        
        return True
```

def clean_text(text: str) -> str:
    """清理文本内容"""
    # 移除页眉页脚（重复内容）
    lines = text.split('\n')
    cleaned_lines = []
    
    for i, line in enumerate(lines):
        # 跳过单独的数字页码
        if line.strip().isdigit() and len(line.strip()) < 5:
            continue
        
        # 跳过重复行
        if i > 0 and line == lines[i-1]:
            continue
        
        cleaned_lines.append(line)
    
    text = '\n'.join(cleaned_lines)
    
    # 移除多余空白
    text = re.sub(r'\n{3,}', '\n\n', text)
    text = re.sub(r' {2,}', ' ', text)
    
    return text
```

def parse_docx_advanced(file_path: str) -> str:
    """高级 DOCX 解析"""
    from docx import Document
    from docx.oxml.text.paragraph import CT_P
    from docx.oxml.table import CT_Tbl
    
    doc = Document(file_path)
    contents = []
    
    for element in doc.element.body:
        if isinstance(element, CT_P):
            # 段落处理
            para = next((p for p in doc.paragraphs if p._element == element), None)
            if para and para.text.strip():
                contents.append(para.text.strip())
        
        elif isinstance(element, CT_Tbl):
            # 表格处理
            table = next((t for t in doc.tables if t._element == element), None)
            if table:
                for row in table.rows:
                    row_cells = []
                    for cell in row.cells:
                        if cell.text.strip():
                            row_cells.append(cell.text.strip())
                    if row_cells:
                        contents.append(" | ".join(row_cells))
    
    return "\n\n".join(contents)

class DocxParser:
    """DOCX 解析器（带错误处理）"""
    
    @staticmethod
    def safe_parse(file_path: str) -> dict:
        """安全解析 DOCX"""
        try:
            from docx import Document
            from zipfile import BadZipFile
            
            doc = Document(file_path)
            texts = []
            
            # 处理段落
            for para in doc.paragraphs:
                try:
                    if para.text.strip():
                        texts.append(para.text.strip())
                except Exception as e:
                    continue
            
            # 处理表格
            for table in doc.tables:
                try:
                    for row in table.rows:
                        for cell in row.cells:
                            if cell.text.strip():
                                texts.append(cell.text.strip())
                except Exception as e:
                    continue
            
            return {
                "success": True,
                "content": "\n\n".join(texts),
                "para_count": len(texts)
            }
        
        except BadZipFile:
            return {
                "success": False,
                "error": "文件格式错误，不是有效的 DOCX 文件"
            }
        except Exception as e:
            return {
                "success": False,
                "error": f"解析失败: {str(e)}"
            }
    
    @classmethod
    def parse_with_fallback(cls, file_path: str) -> Tuple[str, int]:
        """带备用方案的解析"""
        result = cls.safe_parse(file_path)
        
        if result["success"]:
            return result["content"], result["para_count"]
        
        # 备用方案：使用 ZIP 直接提取 XML
        return cls.parse_as_zip(file_path)
    
    @staticmethod
    def parse_as_zip(file_path: str) -> Tuple[str, int]:
        """作为 ZIP 解析 DOCX"""
        import zipfile
        import xml.etree.ElementTree as ET
        
        with zipfile.ZipFile(file_path, 'r') as zf:
            with zf.open('word/document.xml') as xml_file:
                tree = ET.parse(xml_file)
                root = tree.getroot()
                
                # 提取文本
                ns = {'w': 'http://schemas.openxmlformats.org/wordprocessingml/2006/main'}
                texts = []
                
                for elem in root.iter():
                    if elem.tag.endswith('}t') and elem.text:
                        texts.append(elem.text)
                
                return " ".join(texts), len(texts)

### Q16: 如何解析 Markdown 文件？

```python
import re

def parse_markdown(file_path: str) -> Tuple[str, int]:
    """解析 Markdown 文件"""
    with open(file_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    lines = content.split('\n')
    processed_lines = []
    
    for line in lines:
        # 移除图片（保留 alt 文本）
        line = re.sub(r'!\[.*?\]\(.*?\)', '', line)
        
        # 处理链接（保留文本）
        line = re.sub(r'\[(.*?)\]\(.*?\)', r'\1', line)
        
        # 移除代码块标记
        if line.strip().startswith('```'):
            continue
        
        # 移除 HTML 标签
        line = re.sub(r'<[^>]+>', '', line)
        
        processed_lines.append(line)
    
    result = '\n'.join(processed_lines)
    
    # 移除多余空行
    result = re.sub(r'\n{3,}', '\n\n', result)
    
    return result, len(processed_lines)

# 项目中的应用
class FileParser:

            content = f.read()
        
        lines = content.split('\n')
        processed = []

            line = re.sub(r'!\[.*?\]\(.*?\)', '', line)  # 图片
            line = re.sub(r'\[(.*?)\]\(.*?\)', r'\1', line)  # 链接
            line = line.replace('```', '')  # 代码块

        result = '\n'.join(processed)

def parse_txt(file_path: str) -> Tuple[str, int]:
    """解析 TXT 文件"""
    # 尝试不同编码
    encodings = ['utf-8', 'gbk', 'gb2312', 'latin-1']
    content = None
    
    for encoding in encodings:
        try:
            with open(file_path, 'r', encoding=encoding) as f:
                content = f.read()
            break
        except UnicodeDecodeError:
            continue
    
    if content is None:
        # 最后兜底
        with open(file_path, 'r', encoding='utf-8', errors='replace') as f:
            content = f.read()
    
    # 规范化
    content = content.replace('\r\n', '\n').replace('\r', '\n')
    
    line_count = len(content.split('\n'))
    
    return content.strip(), line_count

def chunk_large_file(file_path: str, file_type: str, max_size: int = 1024 * 1024) -> list:
    """分块处理大文件"""
    from pathlib import Path
    
    file_size = Path(file_path).stat().st_size
    
    if file_size <= max_size:
        # 小文件直接处理
        return [(0, None)]
    
    # 大文件分块
    chunks = []
    chunk_size = max_size // 2  # 每块大小
    
    if file_type == "txt" or file_type == "md":
        with open(file_path, 'r', encoding='utf-8') as f:
            offset = 0
            while True:
                f.seek(offset)
                content = f.read(chunk_size)
                if not content:
                    break
                
                chunks.append((offset, content))
                offset += chunk_size - 500  # 重叠处理
    
    return chunks

def merge_chunks(chunks: list) -> str:
    """合并处理后的块"""
    merged = []
    
    for offset, content in chunks:
        if content:
            merged.append(content)
    
    return '\n\n'.join(merged)
```

        }

            result["quality"] = "poor"
            result["issues"].append("内容过短，可能提取失败")
        
        # 检查字符比例
        chinese_chars = sum(1 for c in content if '\u4e00' <= c <= '\u9fff')
        total_chars = len(content.replace(' ', ''))
        
        if total_chars > 0:
            chinese_ratio = chinese_chars / total_chars
            
            if chinese_ratio > 0.9:
                result["issues"].append("几乎全是中文字符")
            elif chinese_ratio < 0.01:
                result["issues"].append("缺少中文字符")
        
        # 检查乱码
        unusual_chars = sum(1 for c in content if ord(c) > 0xFFFD)
        if unusual_chars > len(content) * 0.1:
            result["quality"] = "poor"
            result["issues"].append("可能存在乱码")
        
        # 检查重复
        lines = content.split('\n')
        if len(set(lines)) < len(lines) * 0.3:
            result["issues"].append("内容重复度高")
        
        return result
    
    @staticmethod
    def report_quality(validation_result: dict) -> str:

        elif validation_result["quality"] == "poor":

class FileParser:
    """统一文件解析器"""
    
    @staticmethod
    def calculate_hash(file_path: str) -> str:
        """计算文件哈希"""
        import hashlib
        
        hash_md5 = hashlib.md5()
        with open(file_path, "rb") as f:
            for chunk in iter(lambda: f.read(4096), b""):
                hash_md5.update(chunk)
        return hash_md5.hexdigest()
    
    @classmethod
    def parse_file(cls, file_path: str, file_type: str) -> Tuple[str, int]:
        """根据文件类型解析"""
        parsers = {
            "pdf": cls.parse_pdf,
            "md": cls.parse_markdown,
            "txt": cls.parse_txt,
            "docx": cls.parse_docx,
        }
        
        parser = parsers.get(file_type.lower())
        if not parser:
            raise ValueError(f"不支持的文件类型: {file_type}")
        
        return parser(file_path)
    
    @staticmethod
    def get_file_type(filename: str) -> Optional[str]:
        """获取文件类型"""
        from pathlib import Path
        ext = Path(filename).suffix.lower().lstrip(".")
        
        type_mapping = {
            "pdf": "pdf",
            "md": "md",
            "markdown": "md",
            "txt": "txt",
            "docx": "docx",
        }
        
        return type_mapping.get(ext)

    """提取表格内容"""
    from docx import Document
    
    doc = Document(docx_path)
    tables = []
    
    for table in doc.tables:
        rows = []
        for row in table.rows:
            cells = [cell.text.strip() for cell in row.cells]
            rows.append(cells)
        tables.append(rows)
    
    # 转换为分隔文本
