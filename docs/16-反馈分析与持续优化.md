### 3.2 数据模型

```python
# models/feedback.py (扩展)
class FeedbackAnalysis(Base):
    """反馈分析表"""
    __tablename__ = "feedback_analysis"
    
    id = Column(BigInteger, primary_key=True, autoincrement=True, comment="主键ID")
    qa_log_id = Column(BigInteger, ForeignKey("qa_logs.id"), nullable=False, comment="问答日志ID")
    
    # 反馈信息
    feedback_type = Column(String(20), nullable=False, comment="反馈类型: accurate/inaccurate/unanswerable/partially_accurate")
    rating = Column(Integer, comment="评分 1-5")
    user_feedback_text = Column(Text, comment="用户反馈文本")
    
    # 分析结果
    issue_type = Column(String(50), comment="问题类型: low_recall/low_precision/irrelevant_response/factual_error/no_context")
    root_cause = Column(Text, comment="根本原因分析")
    suggestions = Column(Text, comment="优化建议")
    
    # 状态
    handled_status = Column(String(20), default="pending", comment="处理状态: pending/approved/rejected/applied")
    handled_by = Column(String(50), comment="处理人")
    handled_at = Column(DateTime, comment="处理时间")
    
    # 元数据
    created_at = Column(DateTime, default=datetime.utcnow, comment="创建时间")
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow, comment="更新时间")

class OptimizationRule(Base):
    """优化规则表"""
    __tablename__ = "optimization_rule"
    
    id = Column(BigInteger, primary_key=True, autoincrement=True, comment="主键ID")
    rule_name = Column(String(100), nullable=False, unique=True, comment="规则名称")
    rule_type = Column(String(50), nullable=False, comment="规则类型: cleaning/retrieval/chunking")
    
    # 规则配置
    rule_config = Column(JSON, comment="规则配置 JSON")
    trigger_condition = Column(JSON, comment="触发条件 JSON")
    
    # 规则元信息
    description = Column(Text, comment="规则描述")
    priority = Column(Integer, default=0, comment="优先级")
    enabled = Column(Boolean, default=True, comment="是否启用")
    
    # 状态
    status = Column(String(20), default="draft", comment="状态: draft/active/deprecated")
    approved_by = Column(String(50), comment="审核人")
    approved_at = Column(DateTime, comment="审核时间")
    
    # 效果追踪
    trigger_count = Column(Integer, default=0, comment="触发次数")
    success_count = Column(Integer, default=0, comment="成功次数")
    last_triggered_at = Column(DateTime, comment="最后触发时间")
    
    created_at = Column(DateTime, default=datetime.utcnow, comment="创建时间")
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow, comment="更新时间")
```

### 5.2 问题类型分类


| 类型                  | 说明         |
| ------------------- | ---------- |
| low_recall          | 召回不足，答案不完整 |
| low_precision       | 精度不足，答案不相关 |
| irrelevant_response | 答案跑题       |
| factual_error       | 事实错误       |
| no_context          | 缺乏上下文      |


### 5.3 验证命令

```bash
# 1. 提交反馈
curl -X POST http://localhost:8011/api/v1/qa/feedback \
  -H "Content-Type: application/json" \
  -d '{
    "qa_log_id": 1,
    "feedback_type": "inaccurate",
    "rating": 2,
    "user_feedback_text": "答案中没有提到关键信息"
  }'

# 2. 查询反馈分析
curl http://localhost:8011/api/v1/feedback/analysis/1

# 3. 创建优化规则
curl -X POST http://localhost:8011/api/v1/feedback/rule \
  -H "Content-Type: application/json" \
  -d '{
    "rule_name": "低召回优化规则1",
    "rule_type": "retrieval",
    "rule_config": {"vector_weight": 0.7, "keyword_weight": 0.3},
    "trigger_condition": {"issue_type": "low_recall", "count_threshold": 5}
  }'

# 4. 审核规则
curl -X PUT http://localhost:8011/api/v1/feedback/rule/1/approve

# 5. 查询规则列表
curl http://localhost:8011/api/v1/feedback/rules?rule_type=retrieval
```
