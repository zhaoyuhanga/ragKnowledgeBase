## 二、详细设计

### 2.1 QA 日志表

```sql
CREATE TABLE `qa_logs` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT '日志主键ID',
  `log_id` varchar(64) NOT NULL COMMENT '日志唯一ID',
  `query` text NOT NULL COMMENT '用户问题',
  `answer` text COMMENT '生成答案',
  `rewritten_queries` json DEFAULT NULL COMMENT '改写后的查询',
  `context_chunks` json DEFAULT NULL COMMENT '检索到的Chunk列表',
  `chunk_scores` json DEFAULT NULL COMMENT '各Chunk得分',
  `final_answer` text COMMENT '最终答案',
  `answer_sources` json DEFAULT NULL COMMENT '答案来源',
  `user_id` bigint DEFAULT NULL COMMENT '用户ID',
  `session_id` varchar(64) DEFAULT NULL COMMENT '会话ID',
  `feedback` tinyint DEFAULT NULL COMMENT '用户反馈：1-满意 0-不满意 null-未反馈',
  `feedback_reason` text COMMENT '反馈原因',
  `feedback_at` datetime DEFAULT NULL COMMENT '反馈时间',
  `cost_ms` int DEFAULT NULL COMMENT '耗时（毫秒）',
  `created_at` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_log_id` (`log_id`),
  KEY `idx_user_id` (`user_id`),
  KEY `idx_session_id` (`session_id`),
  KEY `idx_feedback` (`feedback`),
  KEY `idx_created_at` (`created_at`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='问答日志表';
```

### 2.2 反馈服务

```python
class FeedbackService:
    """反馈服务"""

    def submit_feedback(self, log_id: str, feedback: int, reason: str = None) -> bool:
        """提交用户反馈"""
        # 1. 更新 QA 日志
        self.qa_log_repo.update_feedback(log_id, feedback, reason)

        # 2. 分析反馈
        if feedback == 0:  # 不满意
            self._analyze_negative_feedback(log_id)

        return True

    def _analyze_negative_feedback(self, log_id: str):
        """分析负面反馈"""
        log = self.qa_log_repo.get(log_id)

        # 1. 分析检索问题
        if self._is_retrieval_issue(log):
            self._suggest_retrieval_improvement(log)

        # 2. 分析生成问题
        if self._is_generation_issue(log):
            self._suggest_generation_improvement(log)

        # 3. 记录分析结果
        self.feedback_analysis_repo.save({
            "log_id": log_id,
            "issue_type": self._classify_issue(log),
            "suggestions": []
        })

    def _classify_issue(self, log) -> str:
        """分类问题类型"""
        # 此处预留问题分类逻辑
        return "retrieval"  # 或 "generation" 或 "both"
```
