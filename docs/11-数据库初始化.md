## 三、完整建表SQL

### 3.1 初始化脚本

```sql
-- ================================================
-- RAG知识库系统 - MySQL初始化脚本
-- 说明：创建数据库和所有表结构
-- 版本：1.0.0
-- 日期：2026-05-23
-- ================================================

-- 创建数据库（如果不存在）
CREATE DATABASE IF NOT EXISTS rag_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE rag_db;

-- ================================================
-- 文档主表
-- 存储文档基本信息
-- ================================================
CREATE TABLE IF NOT EXISTS documents (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '主键ID',
    tenant_id BIGINT NOT NULL DEFAULT 1 COMMENT '租户ID',
    document_name VARCHAR(500) NOT NULL COMMENT '文档名称',
    file_type VARCHAR(50) COMMENT '文件类型：docx/doc/pdf/txt/md/html/xlsx/png/jpg/jpeg',
    file_hash VARCHAR(64) COMMENT '文件MD5哈希值',
    file_size BIGINT COMMENT '文件大小（字节）',
    status VARCHAR(50) DEFAULT 'pending' COMMENT '状态：pending/parsing/cleaning/chunking/embedding/indexing/completed/failed',
    current_version_id BIGINT COMMENT '当前版本ID',
    total_versions INT DEFAULT 1 COMMENT '版本总数',
    total_pages INT DEFAULT 0 COMMENT '总页数',
    total_chunks INT DEFAULT 0 COMMENT '总Chunk数',
    business_id VARCHAR(100) COMMENT '业务归属ID',
    business_name VARCHAR(100) COMMENT '业务归属名称',
    creator_id BIGINT COMMENT '创建人ID',
    creator_name VARCHAR(100) COMMENT '创建人姓名',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    updater_id BIGINT COMMENT '更新人ID',
    is_deleted TINYINT NOT NULL DEFAULT 0 COMMENT '是否删除：0-否 1-是',
    INDEX idx_documents_tenant (tenant_id),
    INDEX idx_documents_status (status),
    INDEX idx_documents_hash (file_hash),
    INDEX idx_documents_business (business_id),
    INDEX idx_documents_created (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='文档主表';

-- ================================================
-- 文档版本表
-- 存储文档版本信息，支持多版本管理
-- ================================================
CREATE TABLE IF NOT EXISTS document_versions (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '版本主键ID',
    document_id BIGINT NOT NULL COMMENT '文档ID',
    version_number INT NOT NULL DEFAULT 1 COMMENT '版本号',
    file_hash VARCHAR(64) NOT NULL COMMENT '文件MD5哈希值',
    file_name VARCHAR(255) NOT NULL COMMENT '原始文件名',
    storage_path VARCHAR(1000) COMMENT '存储路径',
    file_size BIGINT COMMENT '文件大小（字节）',
    mime_type VARCHAR(100) COMMENT 'MIME类型',
    storage_type VARCHAR(20) DEFAULT 'local' COMMENT '存储类型：local/oss/s3',
    status VARCHAR(50) DEFAULT 'pending' COMMENT '状态：pending/processing/completed/failed',
    parse_status VARCHAR(20) COMMENT '解析状态：pending/processing/completed/failed',
    parse_progress INT DEFAULT 0 COMMENT '解析进度（百分比）',
    parse_confidence DECIMAL(5,4) COMMENT '解析置信度',
    total_pages INT DEFAULT 0 COMMENT '总页数',
    total_elements INT DEFAULT 0 COMMENT '解析元素总数',
    is_active TINYINT NOT NULL DEFAULT 1 COMMENT '是否激活：0-否 1-是',
    uploader_id BIGINT COMMENT '上传人ID',
    uploader_name VARCHAR(100) COMMENT '上传人姓名',
    uploaded_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '上传时间',
    parsed_at DATETIME COMMENT '解析完成时间',
    error_message TEXT COMMENT '错误信息',
    creator_id BIGINT COMMENT '创建人ID',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    INDEX idx_versions_document (document_id),
    INDEX idx_versions_hash (file_hash),
    INDEX idx_versions_status (status),
    INDEX idx_versions_active (is_active)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='文档版本表';

-- ================================================
-- 文档元素表
-- 存储解析后的结构化元素
-- ================================================
CREATE TABLE IF NOT EXISTS document_elements (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '元素主键ID',
    document_id BIGINT NOT NULL COMMENT '文档ID',
    version_id BIGINT NOT NULL COMMENT '版本ID',
    element_id VARCHAR(64) NOT NULL COMMENT '元素唯一ID',
    page_no INT COMMENT '页码（Word可为空）',
    page_start INT COMMENT '起始页（跨页元素）',
    page_end INT COMMENT '结束页',
    element_type VARCHAR(20) NOT NULL COMMENT '元素类型：title/paragraph/table/image/chart/list/header/footer/code',
    text_content TEXT COMMENT '原始文本内容',
    enhanced_content TEXT COMMENT '增强文本内容',
    reading_order INT DEFAULT 0 COMMENT '阅读顺序',
    title_level INT COMMENT '标题层级（1-6）',
    title_path VARCHAR(500) COMMENT '标题层级路径',
    parent_path VARCHAR(500) COMMENT '父级路径',
    bbox JSON COMMENT '元素坐标 {x, y, width, height}',
    confidence DECIMAL(5,4) DEFAULT 1.0000 COMMENT '识别置信度（0-1）',
    is_merged TINYINT DEFAULT 0 COMMENT '是否跨页合并',
    table_structure JSON COMMENT '表格结构信息',
    image_description JSON COMMENT '图片描述信息',
    metadata JSON COMMENT '元数据',
    quality_flag VARCHAR(20) DEFAULT 'good' COMMENT '质量标记：good/warning/bad',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    UNIQUE KEY uk_element_id (element_id),
    INDEX idx_elements_document (document_id),
    INDEX idx_elements_version (version_id),
    INDEX idx_elements_page (page_no),
    INDEX idx_elements_type (element_type),
    INDEX idx_elements_order (reading_order),
    INDEX idx_elements_quality (quality_flag)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='文档解析元素表';

-- ================================================
-- 文档块表（Chunk）
-- 存储切分后的文本块
-- ================================================
CREATE TABLE IF NOT EXISTS document_chunks (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT 'Chunk主键ID',
    document_id BIGINT NOT NULL COMMENT '文档ID',
    version_id BIGINT NOT NULL COMMENT '版本ID',
    chunk_index INT NOT NULL COMMENT '块索引',
    content TEXT NOT NULL COMMENT '块内容',
    content_hash VARCHAR(64) COMMENT '内容哈希',
    enhanced_content TEXT COMMENT '增强内容（包含标题上下文等）',
    block_type VARCHAR(50) DEFAULT 'paragraph' COMMENT '块类型：paragraph/table/image/chart/list/code',
    title_path VARCHAR(500) COMMENT '标题路径',
    page_start INT COMMENT '起始页码',
    page_end INT COMMENT '结束页码',
    token_count INT COMMENT 'Token数量',
    quality_score DECIMAL(5,4) COMMENT '质量评分',
    source_type VARCHAR(20) DEFAULT 'local' COMMENT '来源类型：local/ai_generated',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    INDEX idx_chunks_document (document_id),
    INDEX idx_chunks_version (version_id),
    INDEX idx_chunks_hash (content_hash),
    INDEX idx_chunks_type (block_type),
    FULLTEXT INDEX ft_chunks_content (content)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='文档块表';

-- ================================================
-- 导入任务表
-- 存储导入任务信息
-- ================================================
CREATE TABLE IF NOT EXISTS import_tasks (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '任务主键ID',
    task_id VARCHAR(64) NOT NULL COMMENT '任务唯一ID（UUID）',
    document_id BIGINT COMMENT '关联文档ID',
    version_id BIGINT COMMENT '关联版本ID',
    task_type VARCHAR(20) DEFAULT 'import' COMMENT '任务类型：import/parse/clean/chunk/embed/index',
    task_status VARCHAR(20) NOT NULL DEFAULT 'pending' COMMENT '任务状态：pending/running/completed/failed/retry',
    priority INT DEFAULT 5 COMMENT '优先级：1-最高 5-普通',
    progress INT DEFAULT 0 COMMENT '进度（百分比）',
    retry_count INT DEFAULT 0 COMMENT '重试次数',
    max_retry INT DEFAULT 3 COMMENT '最大重试次数',
    error_type VARCHAR(50) COMMENT '错误类型',
    error_message TEXT COMMENT '错误信息',
    error_detail TEXT COMMENT '错误详情',
    started_at DATETIME COMMENT '开始时间',
    completed_at DATETIME COMMENT '完成时间',
    cost_seconds INT COMMENT '耗时（秒）',
    payload JSON COMMENT '任务参数',
    result JSON COMMENT '任务结果',
    creator_id BIGINT COMMENT '创建人ID',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    UNIQUE KEY uk_task_id (task_id),
    INDEX idx_tasks_document (document_id),
    INDEX idx_tasks_version (version_id),
    INDEX idx_tasks_status (task_status),
    INDEX idx_tasks_created (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='导入任务表';

-- ================================================
-- 关键词倒排索引表
-- 存储Chunk的关键词索引
-- ================================================
CREATE TABLE IF NOT EXISTS chunk_keyword_index (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '主键ID',
    chunk_id BIGINT NOT NULL COMMENT 'Chunk ID',
    term VARCHAR(128) NOT NULL COMMENT '分词词项',
    field VARCHAR(20) NOT NULL DEFAULT 'content' COMMENT '字段：content/title/enhanced',
    tf INT NOT NULL DEFAULT 1 COMMENT '词频',
    idf DECIMAL(10,4) NOT NULL DEFAULT 0 COMMENT '逆文档频率',
    position INT COMMENT '词项位置',
    weight DECIMAL(5,4) NOT NULL DEFAULT 1.0000 COMMENT '权重',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    UNIQUE KEY uk_chunk_term_field (chunk_id, term, field),
    INDEX idx_index_term (term),
    INDEX idx_index_chunk (chunk_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='关键词倒排索引表';

-- ================================================
-- 问答日志表
-- 存储问答历史记录
-- ================================================
CREATE TABLE IF NOT EXISTS qa_logs (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '问答日志ID',
    tenant_id BIGINT NOT NULL DEFAULT 1 COMMENT '租户ID',
    user_id BIGINT COMMENT '用户ID',
    session_id VARCHAR(100) COMMENT '会话ID',
    question TEXT NOT NULL COMMENT '用户问题',
    answer TEXT COMMENT '生成的答案',
    references_json TEXT COMMENT '引用的Chunk JSON',
    avg_retrieval_score DECIMAL(5,4) COMMENT '平均检索分数',
    reference_count INT DEFAULT 0 COMMENT '引用数量',
    retrieval_time_ms INT COMMENT '检索耗时（毫秒）',
    rerank_time_ms INT COMMENT '重排耗时（毫秒）',
    generation_time_ms INT COMMENT '生成耗时（毫秒）',
    total_time_ms INT COMMENT '总耗时（毫秒）',
    feedback TINYINT COMMENT '反馈：1-满意 0-不满意',
    feedback_reason TEXT COMMENT '反馈原因',
    quality_score INT COMMENT '质量评分 1-5',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    INDEX idx_qa_tenant (tenant_id),
    INDEX idx_qa_user (user_id),
    INDEX idx_qa_session (session_id),
    INDEX idx_qa_created (created_at),
    INDEX idx_qa_feedback (feedback),
    FULLTEXT INDEX ft_qa_question (question)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='问答日志表';

-- ================================================
-- 反馈分析表
-- 存储用户反馈分析结果
-- ================================================
CREATE TABLE IF NOT EXISTS feedback_analysis (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '分析ID',
    qa_log_id BIGINT NOT NULL COMMENT '问答日志ID',
    issue_type VARCHAR(50) COMMENT '问题类型：retrieval/generation/both',
    issue_category VARCHAR(100) COMMENT '问题分类',
    issue_description TEXT COMMENT '问题描述',
    involved_chunks TEXT COMMENT '涉及Chunk ID列表（JSON）',
    retrieval_score INT COMMENT '检索质量评分',
    generation_score INT COMMENT '生成质量评分',
    suggestions TEXT COMMENT '改进建议列表（JSON）',
    suggestion_type VARCHAR(100) COMMENT '建议类型',
    handled_status INT DEFAULT 0 COMMENT '处理状态：0-未处理 1-已处理 2-已忽略',
    handler_id BIGINT COMMENT '处理人ID',
    handled_at DATETIME COMMENT '处理时间',
    handler_remark TEXT COMMENT '处理备注',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    INDEX idx_analysis_qa (qa_log_id),
    INDEX idx_analysis_type (issue_type),
    INDEX idx_analysis_status (handled_status)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='反馈分析表';

-- ================================================
-- 优化规则表
-- 存储系统优化规则
-- ================================================
CREATE TABLE IF NOT EXISTS optimization_rules (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '规则ID',
    rule_name VARCHAR(200) NOT NULL COMMENT '规则名称',
    rule_type VARCHAR(50) NOT NULL COMMENT '规则类型：cleaning/chunking/retrieval/rerank',
    rule_config TEXT COMMENT '规则配置JSON',
    trigger_condition TEXT COMMENT '触发条件JSON',
    priority INT DEFAULT 2 COMMENT '优先级：1-高 2-中 3-低',
    enabled TINYINT DEFAULT 1 COMMENT '启用状态：0-禁用 1-启用',
    description TEXT COMMENT '规则描述',
    applicable_scope TEXT COMMENT '应用范围JSON',
    expected_effect TEXT COMMENT '预期效果',
    actual_effect TEXT COMMENT '实际效果评估',
    creator_id BIGINT COMMENT '创建人ID',
    updater_id BIGINT COMMENT '最后修改人ID',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    INDEX idx_rules_type (rule_type),
    INDEX idx_rules_enabled (enabled),
    INDEX idx_rules_priority (priority)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='优化规则表';

-- ================================================
-- 清洗规则表
-- 存储文本清洗规则
-- ================================================
CREATE TABLE IF NOT EXISTS cleaning_rules (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '规则ID',
    rule_name VARCHAR(200) NOT NULL COMMENT '规则名称',
    rule_type VARCHAR(50) NOT NULL COMMENT '规则类型：regex_delete/regex_replace/structure_delete/quality_control/desensitization',
    rule_config TEXT NOT NULL COMMENT '规则配置JSON',
    scope_type VARCHAR(50) DEFAULT 'global' COMMENT '适用范围类型：global/business/document_type',
    scope_value VARCHAR(500) COMMENT '适用范围值',
    priority INT DEFAULT 2 COMMENT '优先级（数字越小优先级越高）',
    enabled TINYINT DEFAULT 1 COMMENT '是否启用：0-禁用 1-启用',
    description TEXT COMMENT '规则描述',
    creator_id BIGINT COMMENT '创建人ID',
    updater_id BIGINT COMMENT '最后修改人ID',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    INDEX idx_clean_type (rule_type),
    INDEX idx_clean_enabled (enabled),
    INDEX idx_clean_priority (priority)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='清洗规则表';

-- ================================================
-- 解析质量日志表
-- 存储解析质量检查结果
-- ================================================
CREATE TABLE IF NOT EXISTS parse_quality_logs (
    id BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '日志ID',
    document_id BIGINT NOT NULL COMMENT '文档ID',
    version_id BIGINT NOT NULL COMMENT '版本ID',
    page_no INT COMMENT '页码',
    element_id VARCHAR(64) COMMENT '元素ID',
    check_type VARCHAR(50) NOT NULL COMMENT '检查类型：ocr/low_confidence/layout/encoding',
    quality_flag VARCHAR(20) NOT NULL COMMENT '质量标记：good/warning/bad',
    confidence DECIMAL(5,4) COMMENT '置信度',
    issue_description TEXT COMMENT '问题描述',
    suggestion TEXT COMMENT '修复建议',
    resolved TINYINT NOT NULL DEFAULT 0 COMMENT '是否已解决：0-否 1-是',
    resolved_at DATETIME COMMENT '解决时间',
    resolved_by BIGINT COMMENT '解决人ID',
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    INDEX idx_quality_document (document_id),
    INDEX idx_quality_version (version_id),
    INDEX idx_quality_page (page_no),
    INDEX idx_quality_flag (quality_flag),
    INDEX idx_quality_resolved (resolved)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='解析质量日志表';
```
