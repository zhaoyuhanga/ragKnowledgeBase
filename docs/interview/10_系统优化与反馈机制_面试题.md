| 指标 | 计算方法 | 说明 |
|------|----------|------|
| NPS | 推荐者-贬损者 | 净推荐值 |
| CSAT | 满意数/总数 | 满意度 |
| CES | 努力程度评分 | 便捷度 |

**评估实现：**

```python
def calculate_satisfaction(feedbacks):
    # 1. 基础指标
    positive_rate = sum(1 for f in feedbacks if f.positive) / len(feedbacks)
    
    # 2. 满意度评分
    satisfaction = positive_rate * 100
    
    # 3. NPS计算
    promoters = sum(1 for f in feedbacks if f.score >= 9)
    detractors = sum(1 for f in feedbacks if f.score <= 6)
    nps = (promoters - detractors) / len(feedbacks) * 100
    
    return {
        'satisfaction': satisfaction,
        'nps': nps,
        'trend': calculate_trend(feedbacks)
    }
```

```python
def evaluate_stability(duration_hours=24):
    start_time = time.time()
    metrics = {
        'total_requests': 0,
        'failed_requests': 0,
        'latencies': [],
        'errors': []
    }
    
    while time.time() - start_time < duration_hours * 3600:
        try:
            result = make_request()
            metrics['latencies'].append(result.latency)
        except Exception as e:
            metrics['failed_requests'] += 1
            metrics['errors'].append(str(e))
        
        metrics['total_requests'] += 1
    
    return {
        'availability': 1 - metrics['failed_requests'] / metrics['total_requests'],
        'error_rate': metrics['failed_requests'] / metrics['total_requests'],
        'latency_p99': percentile(metrics['latencies'], 99)
    }
```

```python
class ContinuousEvaluator:
    def __init__(self):

    async def run(self):
```
