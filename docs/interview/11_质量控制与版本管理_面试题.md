**答案要点**：

**版本管理设计：**

```python
class DocumentVersionManager:
    async def create_version(self, doc_id, new_content):
        # 1. 计算内容hash
        content_hash = calculate_hash(new_content)
        
        # 2. 获取当前版本
        current = await self.get_current_version(doc_id)
        
        # 3. 比较hash
        if current and current.hash == content_hash:
            return current  # 内容未变
        
        # 4. 创建新版本
        version = await self.create({
            'doc_id': doc_id,
            'version': current.version + 1 if current else 1,
            'hash': content_hash,
            'content': new_content,
            'created_at': datetime.now()
        })
        
        # 5. 更新文档当前版本
        await self.update_document_current(doc_id, version.id)
        
        # 6. 触发重新索引
        await self.trigger_reindex(doc_id, version.id)
        
        return version
```

class ModelVersionManager:

```python
class DataMigrationManager:
    def __init__(self):
        self.migrations = []
    
    def register_migration(self, version, up_func, down_func):
        """注册迁移脚本"""
        self.migrations.append({
            'version': version,
            'up': up_func,
            'down': down_func
        })
    
    async def migrate_to(self, target_version):
        """迁移到目标版本"""
        current = self.get_current_version()
        
        if target_version > current:
            # 升级
            for v in range(current + 1, target_version + 1):
                await self.migrations[v].up()
        else:
            # 降级
            for v in range(current, target_version, -1):
                await self.migrations[v].down()
```
