```python
# 检索指标
def calculate_recall_at_k(retrieved, relevant, k=10):
    """Recall@K"""
    retrieved_topk = set(retrieved[:k])
    relevant_set = set(relevant)
    return len(retrieved_topk & relevant_set) / len(relevant_set)

def calculate_mrr(retrieved, relevant):
    """MRR"""
    for i, doc in enumerate(retrieved, 1):
        if doc in relevant:
            return 1 / i
    return 0

# 生成指标
def evaluate_answer(generated, expected):
```

```python
            for query in queries:
                answer = await llm.ask(query, context=doc.content)
                
                training_data.append({
                    'query': query,
                    'answer': answer,
                    'document_id': doc.id,
                    'positive_chunks': doc.relevant_chunks
                })
        
        return training_data
    
    async def distill(self, training_data):
        """蒸馏到小模型"""
        # 1. 微调Embedding
        self.small_embedding.fit(training_data)
        
        # 2. 微调排序模型
        self.small_reranker.fit(training_data)
        
        # 3. 评估效果
        metrics = self.evaluate(self.small_model)
        return metrics
```

### Q29: 设计一个支持联邦学习的RAG系统

**题目类型**：综合设计类

**问题描述**：多个机构的数据不能直接共享，如何使用联邦学习提升RAG效果？

**答案要点**：

**联邦学习架构：**

```
┌─────────┐  ┌─────────┐  ┌─────────┐
│ 机构A   │  │ 机构B   │  │ 机构C   │
│  │     │  │  │     │  │  │     │
│  ↓     │  │  ↓     │  │  ↓     │
│ 本地训练 │  │ 本地训练 │  │ 本地训练 │
│  ↓     │  │  ↓     │  │  ↓     │
│  参数   │  │  参数   │  │  参数   │
└────┬────┘  └────┬────┘  └────┬────┘
     └────────────┴────────────┘
              ↓
      参数聚合服务器
              ↓
     ┌────────┴────────┐
     ↓                 ↓
更新后的全局模型  更新后的全局模型
```

**实现示例：**

```python
class FederatedRAG:
    async def federated_train(self, clients):
        """联邦训练"""
        global_model = self.model
        
        for round in range(num_rounds):
            # 1. 分发全局模型
            for client in clients:
                await client.update_model(global_model)
            
            # 2. 本地训练
            client_updates = []
            for client in clients:
                update = await client.local_train()
                client_updates.append(update)
            
            # 3. 聚合参数 (FedAvg)
            global_model = self.aggregate(client_updates)
        
        return global_model
    
    def aggregate(self, updates):
        """参数聚合"""
        # FedAvg算法
        total_weight = sum(u.weight for u in updates)
        
        aggregated = {}
        for key in updates[0].params.keys():
            aggregated[key] = sum(
                u.params[key] * u.weight / total_weight
                for u in updates
            )
        
        return aggregated
```

```python
class RAGEvaluator:
    def __init__(self):
        self.metrics = {
            # 检索指标
            'retrieval': {
                'recall@1': lambda r: calculate_recall(r, k=1),
                'recall@5': lambda r: calculate_recall(r, k=5),
                'recall@10': lambda r: calculate_recall(r, k=10),
                'mrr': calculate_mrr,
                'ndcg': calculate_ndcg
            },
            # 生成指标
            'generation': {
                'accuracy': evaluate_accuracy,
                'completeness': evaluate_completeness,
                'hallucination_rate': evaluate_hallucination,
                'relevance': evaluate_relevance
            },
            # 业务指标
            'business': {
                'user_satisfaction': calculate_satisfaction,
                'solution_rate': calculate_solution_rate,
                'escalation_rate': calculate_escalation_rate
            },
            # 性能指标
            'performance': {
                'latency_p50': calculate_latency_p50,
                'latency_p99': calculate_latency_p99,
                'qps': calculate_qps,
                'error_rate': calculate_error_rate
            }
        }
    
    def evaluate(self, test_set):
        results = {}
        
        for category, metrics in self.metrics.items():
            category_results = {}
            for metric_name, metric_func in metrics.items():
                category_results[metric_name] = metric_func(test_set)
            results[category] = category_results
        
        return results
```
