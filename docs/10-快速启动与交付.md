## 二、快速启动脚本

### 2.1 一键启动脚本 (start-all-local.bat)

```batch
@echo off
chcp 65001 > nul
echo ========================================
echo   RAG 知识库系统 - 本地环境一键启动
echo ========================================
echo.

:: 检查 MySQL
echo [1/6] 检查 MySQL 服务...
sc query MySQL | findstr "RUNNING" > nul
if errorlevel 1 (
    echo   MySQL 未运行，请先启动 MySQL
) else (
    echo   MySQL 已运行
)

:: 检查 Redis
echo [2/6] 检查 Redis 服务...
sc query Redis | findstr "RUNNING" > nul
if errorlevel 1 (
    echo   Redis 未运行，请先启动 Redis
) else (
    echo   Redis 已运行
)

:: 启动 Milvus
echo [3/6] 启动 Milvus 服务...
docker run -d --name milvus-etcd -p 2379:2379 -p 2380:2380 quay.io/coreos/etcd:latest
docker run -d --name milvus-minio -p 9000:9000 -p 9001:9001 minio/minio:latest server /minio/data
docker run -d --name milvus -p 19530:19530 -p 9091:9091 milvusdb/milvus:v2.4.0

:: 启动 RabbitMQ
echo [4/6] 启动 RabbitMQ 服务...
docker run -d --name rabbitmq -p 5672:5672 -p 15672:15672 rabbitmq:3.12-management

:: 启动后端
echo [5/6] 启动后端服务...
cd /d %~dp0backend\src
start "RAG-Backend" cmd /k "python -m uvicorn main:app --host 127.0.0.1 --port 8011 --reload"

:: 启动前端（如果存在）
echo [6/6] 启动前端服务...
cd /d %~dp0frontend
start "RAG-Frontend" cmd /k "npm run dev:local"

echo.
echo ========================================
echo   启动完成！
echo   后端地址: http://localhost:8011
echo   API文档:  http://localhost:8011/docs
echo   Milvus:   http://localhost:9091
echo   RabbitMQ: http://localhost:15672
echo ========================================
echo.
pause
```

### 2.2 Docker Compose 配置

```yaml
# docker-compose.yml
version: '3.8'

services:
  mysql:
    image: mysql:8.0
    container_name: rag-mysql
    environment:
      MYSQL_ROOT_PASSWORD: root
      MYSQL_DATABASE: rag_db
    ports:
      - "3308:3306"
    volumes:
      - mysql_data:/var/lib/mysql
    command: --default-authentication-plugin=mysql_native_password

  redis:
    image: redis:7-alpine
    container_name: rag-redis
    ports:
      - "6379:6379"

  milvus-etcd:
    image: quay.io/coreos/etcd:latest
    container_name: rag-milvus-etcd
    environment:
      ETCD_AUTO_COMPACTION_MODE: revision
      ETCD_AUTO_COMPACTION_RETENTION: "1000"
      ETCD_QUOTA_BACKEND_BYTES: 4294967296
    volumes:
      - etcd_data:/etcd
    command: etcd -advertise-client-urls=http://127.0.0.1:2379 -listen-client-urls http://0.0.0.0:2379 --data-dir /etcd

  milvus-minio:
    image: minio/minio:latest
    container_name: rag-milvus-minio
    environment:
      MINIO_ACCESS_KEY: minioadmin
      MINIO_SECRET_KEY: minioadmin
    volumes:
      - minio_data:/minio/data
    command: minio server /minio/data

  milvus:
    image: milvusdb/milvus:v2.4.0
    container_name: rag-milvus
    environment:
      ETCD_ENDPOINTS: milvus-etcd:2379
      MINIO_ADDRESS: milvus-minio:9000
    ports:
      - "19530:19530"
      - "9091:9091"
    depends_on:
      - milvus-etcd
      - milvus-minio

  rabbitmq:
    image: rabbitmq:3.12-management
    container_name: rag-rabbitmq
    environment:
      RABBITMQ_DEFAULT_USER: guest
      RABBITMQ_DEFAULT_PASS: guest
    ports:
      - "5672:5672"
      - "15672:15672"

volumes:
  mysql_data:
  etcd_data:
  minio_data:
```
