### 3.2 API 设计

```python
# schemas/query_rewrite.py
from pydantic import BaseModel, Field
from typing import List, Optional
from enum import Enum

class RewriteStrategy(str, Enum):
    """改写策略枚举"""
    NORMALIZE = "normalize"      # 仅规范化
    MULTI_QUERY = "multi_query"  # 多查询
    DECOMPOSE = "decompose"       # 子查询分解
    HYDE = "hyde"               # HyDE（预留）
    BACKWARD = "backward"        # 后退提示（预留）

class QueryRewriteRequest(BaseModel):
    """查询改写请求"""
    query: str = Field(description="原始查询")
    strategy: RewriteStrategy = Field(default=RewriteStrategy.MULTI_QUERY, description="改写策略")
    max_queries: int = Field(default=5, description="最大查询数量")
    include_original: bool = Field(default=True, description="是否包含原始查询")

class QueryRewriteResponse(BaseModel):
    """查询改写响应"""
    original_query: str
    rewritten_queries: List[str]
    strategy_used: RewriteStrategy
    processing_time_ms: int
```
