## 二、详细设计

### 2.1 重排序服务

```python
class RerankService:
    """重排序服务"""

    def rerank(self, query: str, candidates: List[Chunk], top_k: int = 10) -> List[RerankedChunk]:
        """Cross-Encoder 重排序"""

        # 1. 构造 query-chunk pairs
        pairs = [(query, chunk.content) for chunk in candidates]

        # 2. 批量重排序
        scores = self.cross_encoder.predict(pairs)

        # 3. 按分数排序
        reranked = []
        for chunk, score in zip(candidates, scores):
            reranked.append({
                **chunk,
                'rerank_score': float(score)
            })

        reranked.sort(key=lambda x: x['rerank_score'], reverse=True)

        # 4. 过滤低分
        reranked = [r for r in reranked if r['rerank_score'] > self.config.min_score]

        return reranked[:top_k]

    def assemble_context(self, chunks: List[RerankedChunk], max_tokens: int = 4000) -> str:
        """上下文组装"""

        context_parts = []
        current_tokens = 0

        for chunk in chunks:
            chunk_tokens = self._estimate_tokens(chunk['content'])

            if current_tokens + chunk_tokens > max_tokens:
                break

            # 添加引用标记
            citation = f"[来源: {chunk['document_name']}, 第{chunk['page_no']}页]"
            context_parts.append(f"{chunk['content']}\n{citation}")

            current_tokens += chunk_tokens

        return "\n\n".join(context_parts)
```

### 2.2 问答服务

```python
class QAService:
    """问答服务"""

    def __init__(self):
        self.prompt_builder = PromptBuilder()

    def ask(self, query: str, context: str, use_citation: bool = True) -> QAResult:
        """问答主入口"""

        # 1. 构造 Prompt
        prompt = self.prompt_builder.build(query, context, use_citation)

        # 2. 调用 LLM
        answer = self.llm_service.generate(prompt)

        # 3. 提取引用
        citations = self._extract_citations(answer)

        return QAResult(
            answer=answer,
            citations=citations
        )

    def build_prompt(self, query: str, context: str, use_citation: bool) -> str:
        """构造 Prompt"""

        template = """你是一个专业的知识库问答助手。请根据以下参考信息回答用户的问题。

参考信息：
{context}

用户问题：{query}

要求：
1. 只根据参考信息回答，不要编造信息
2. 如果参考信息不足以回答，请明确说明
3. 回答要准确、简洁、有条理
{ citation_instruction }

回答："""

        citation_instruction = ""
        if use_citation:
            citation_instruction = "4. 在回答中使用 [来源: ...] 标注信息来源"

        return template.format(
            context=context,
            query=query,
            citation_instruction=citation_instruction
        )
```
