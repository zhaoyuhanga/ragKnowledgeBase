## 二、详细设计

### 2.1 消息格式

```python
class TaskMessage(BaseModel):
    """任务消息"""
    task_id: str                    # UUID
    task_type: TaskType             # parse/clean/chunk/embedding/index
    document_id: int
    version_id: int
    priority: int = 5              # 1-最高，5-普通
    retry_count: int = 0
    created_at: datetime
    payload: dict = {}

class TaskType(str, Enum):
    """任务类型"""
    PARSE = "parse"
    CLEAN = "clean"
    CHUNK = "chunk"
    EMBEDDING = "embedding"
    INDEX = "index"
```

### 2.2 队列消费者

```python
class QueueConsumer:
    """RabbitMQ 队列消费者"""

    def __init__(self):
        self.connection = None
        self.channels = {}
        self.workers = {
            TaskType.PARSE: ParseWorker(),
            TaskType.CLEAN: CleanWorker(),
            TaskType.CHUNK: ChunkWorker(),
            TaskType.EMBEDDING: EmbeddingWorker(),
            TaskType.INDEX: IndexWorker(),
        }

    def start(self):
        """启动消费者"""
        self._connect()
        self._declare_queues()
        self._start_consumers()

    def _start_consumers(self):
        """启动各队列消费者"""
        for task_type, worker in self.workers.items():
            channel = self._get_channel(task_type)
            queue_name = self._get_queue_name(task_type)

            channel.basic_qos(prefetch_count=worker.prefetch_count)
            channel.basic_consume(
                queue=queue_name,
                on_message_callback=self._create_callback(worker)
            )

    def _create_callback(self, worker: BaseWorker):
        """创建消息回调"""

        def callback(ch, method, properties, body):
            try:
                message = TaskMessage.parse_raw(body)
                logger.info(f"收到任务: {message.task_id}")

                # 执行任务
                result = worker.process(message)

                # 确认消息
                ch.basic_ack(delivery_tag=method.delivery_tag)

                # 发布下一阶段任务
                if result.get("next_task"):
                    self._publish_next(result["next_task"])

            except RetryableError as e:
                # 可重试错误，重试
                self._handle_retry(ch, method, body, e)
            except NonRetryableError as e:
                # 不可重试错误，进入死信队列
                self._handle_dead_letter(ch, method, body, e)
            except Exception as e:
                # 其他错误
                logger.error(f"任务执行异常: {e}")
                ch.basic_nack(delivery_tag=method.delivery_tag, requeue=False)

        return callback

    def _handle_retry(self, ch, method, body, error):
        """处理重试"""
        message = TaskMessage.parse_raw(body)

        if message.retry_count < MAX_RETRY_COUNT:
            # 延迟重试
            delay = min(2 ** message.retry_count * 5, 300)  # 延迟 5s, 10s, 20s...
            time.sleep(delay)

            message.retry_count += 1
            ch.basic_publish(
                exchange='',
                routing_key=self._get_queue_name(TaskType(message.task_type)),
                body=message.json()
            )
            ch.basic_ack(delivery_tag=method.delivery_tag)
        else:
            # 超过最大重试次数，进入死信队列
            self._handle_dead_letter(ch, method, body, error)
```

### 2.3 Worker 实现

```python
class ParseWorker(BaseWorker):
    """解析 Worker"""

    prefetch_count = 2

    def process(self, message: TaskMessage) -> dict:
        """执行解析任务"""
        version_id = message.version_id

        # 1. 更新任务状态为运行中
        self._update_task_status(message.task_id, TaskStatus.RUNNING)

        # 2. 执行解析
        parse_service = ParseService()
        elements = parse_service.parse(version_id)

        # 3. 保存元素
        self._save_elements(elements)

        # 4. 更新版本状态
        self._update_version_status(version_id, VersionStatus.PARSED)

        return {
            "status": "success",
            "next_task": {
                "task_type": TaskType.CLEAN,
                "document_id": message.document_id,
                "version_id": version_id,
                "payload": {"element_count": len(elements)}
            }
        }
```
