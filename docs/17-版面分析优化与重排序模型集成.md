#### 3.1.1 栏结构识别算法

```python
def _identify_columns(self, elements: List[DocumentElementModel]) -> List[DocumentElementModel]:
    """
    识别栏结构
    
    基于元素X坐标分布聚类，识别单栏/双栏/多栏布局。
    """
    if not elements:
        return elements
    
    # 1. 提取所有元素的X坐标
    x_coords = []
    for elem in elements:
        if elem.bbox:
            x_coords.append(elem.bbox.x)
            if elem.bbox.width:
                x_coords.append(elem.bbox.x + elem.bbox.width)
    
    if not x_coords:
        return elements
    
    # 2. 使用K-Means聚类识别栏边界
    n_columns = self._estimate_column_count(x_coords)
    
    # 3. 为每个元素分配栏编号
    column_boundaries = self._compute_boundaries(x_coords, n_columns)
```

#### 3.1.2 多栏阅读顺序

```python
def _reorder_by_reading(self, elements: List[DocumentElementModel]) -> List[DocumentElementModel]:
    """
    按阅读顺序重排（支持多栏）
    
    单栏：从上到下、从左到右
    双栏：Z字形阅读（左上→右上→左中→右中...）
    """
    # 1. 按页分组
    page_elements = self._group_by_page(elements)
    
    # 2. 确定栏数
    column_count = self._detect_column_count(page_elements)
    
    # 3. 按阅读顺序排序
    ordered = []
    for page_no in sorted(page_elements.keys()):
        page_items = page_elements[page_no]
        if column_count == 1:
            # 单栏：从上到下
            page_items.sort(key=lambda e: (e.bbox.y if e.bbox else 0))
        else:
            # 多栏：Z字形
            page_items.sort(key=lambda e: (
                e.bbox.y if e.bbox else 0,
                e.bbox.x if e.bbox else 0
            ))
        ordered.extend(page_items)
    
    return ordered
```

### 3.2 Ollama Cross-Encoder 集成

#### 3.2.1 Ollama Rerank API

Ollama 原生不支持 Cross-Encoder，可以使用以下方案：

1. **方案A**：使用 `bge-reranker` 模型（如果 Ollama 支持）
2. **方案B**：使用 Ollama 的通用模型模拟
3. **方案C**：调用远程 Rerank API（如 Jina Reranker）

```python
# ollama_client.py 新增

class OllamaReranker:
    """Ollama 重排序客户端"""
    
    def __init__(self, host: str = "http://localhost:11434"):
        self.host = host
    
    async def rerank(
        self,
        query: str,
        documents: List[str],
        model: str = "bge-reranker"
    ) -> List[float]:
        """
        重排序评分
        
        Args:
            query: 查询文本
            documents: 文档列表
            model: 模型名称
        
        Returns:
            相关性分数列表
        """
        # 方案：使用 Ollama 的 embedding 接口计算相似度
        async with httpx.AsyncClient(timeout=60) as client:
            # 获取查询向量
            query_resp = await client.post(
                f"{self.host}/api/embeddings",
                json={"model": "qwen3-embedding:4b-q6", "prompt": query}
            )
            query_embedding = query_resp.json()["embedding"]
            
            # 获取文档向量
            scores = []
            for doc in documents:
                doc_resp = await client.post(
                    f"{self.host}/api/embeddings",
                    json={"model": "qwen3-embedding:4b-q6", "prompt": doc}
                )
                doc_embedding = doc_resp.json()["embedding"]
                
                # 计算余弦相似度
                score = self._cosine_similarity(query_embedding, doc_embedding)
                scores.append(score)
            
            return scores
    
    def _cosine_similarity(self, a: List[float], b: List[float]) -> float:
        """计算余弦相似度"""
        dot = sum(x * y for x, y in zip(a, b))
        norm_a = sum(x * x for x in a) ** 0.5
        norm_b = sum(x * x for x in b) ** 0.5
        return dot / (norm_a * norm_b) if norm_a and norm_b else 0.0
```
